3. ОПТИМИЗАЦИЯ: КАК УСКОРИТЬ F#
================================================================================

Совет 1: Используйте tail-рекурсию
----------------------------------
// МЕДЛЕННО (обычная рекурсия):
let rec sum n =
    if n <= 0 then 0
    else n + sum (n - 1)

// БЫСТРО (tail-рекурсия):
let sumTail n =
    let rec loop acc n =
        if n <= 0 then acc
        else loop (acc + n) (n - 1)
    loop 0 n

// ⚡ F# оптимизирует tail-рекурсию!

Совет 2: Используйте arrays для числовых данных
-----------------------------------------------
// МЕДЛЕННО (list):
let list = [1..1000000]
List.sum list

// БЫСТРО (array):
let array = [|1..1000000|]
Array.sum array

// ⚡ Arrays быстрее для числовых данных!

Совет 3: Избегайте mutable в hot path
-------------------------------------
// МЕДЛЕННО (mutable):
let mutable sum = 0
for i in 1..1000000 do
    sum <- sum + i

// БЫСТРО (immutable):
let sum = [1..1000000] |> List.sum

// ⚡ F# оптимизирует immutable код!

Совет 4: Используйте inline функции
-----------------------------------
// МЕДЛЕННО (обычная функция):
let add a b = a + b

// БЫСТРО (inline):
[<Inline>]
let add a b = a + b

// ⚡ Inline функции встраиваются!

Совет 5: Используйте Parallel Seq
---------------------------------
// МЕДЛЕННО (последовательно):
seq { 1..1000000 }
|> Seq.map (fun x -> x * 2)
|> Seq.sum

// БЫСТРО (параллельно):
seq { 1..1000000 }
|> Seq.map (fun x -> x * 2)
|> Seq.sumByParallel id

// ⚡ Параллельные вычисления быстрее!
