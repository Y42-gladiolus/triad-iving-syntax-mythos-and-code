3. ОПТИМИЗАЦИЯ: КАК УСКОРИТЬ JULIA
================================================================================

Совет 1: Избегайте глобальных переменных
----------------------------------------
# МЕДЛЕННО (глобальная):
x = 0
for i in 1:1000000
    x += i
end

# БЫСТРО (в функции):
function fast_sum()
    x = 0
    for i in 1:1000000
        x += i
    end
    return x
end

# ⚡ Ускорение: 10-100x!

Совет 2: Используйте конкретные типы
------------------------------------
# МЕДЛЕННО (Any тип):
function slow_function()
    x = 0  # Any
    for i in 1:1000
        x += i
    end
    return x
end

# БЫСТРО (конкретный тип):
function fast_function()
    x::Int = 0
    for i in 1:1000
        x += i
    end
    return x
end

# ⚡ Ускорение: 5-50x!

Совет 3: Векторизация с точкой (.)
----------------------------------
# МЕДЛЕННО (цикл):
result = []
for i in 1:10000
    push!(result, i^2)
end

# БЫСТРО (векторизация):
result = (1:10000) .^ 2

# ⚡ Ускорение: 10-100x!

Совет 4: Предварительное выделение памяти
-----------------------------------------
# МЕДЛЕННО (рост в цикле):
result = []
for i in 1:10000
    push!(result, i)
end

# БЫСТРО (предварительное):
result = Vector{Int}(undef, 10000)
for i in 1:10000
    result[i] = i
end

# ⚡ Ускорение: 5-20x!

Совет 5: Используйте @inbounds и @simd
---------------------------------------
# Обычный цикл:
function sum_array(arr)
    total = 0
    for i in eachindex(arr)
        total += arr[i]
    end
    return total
end

# Оптимизированный:
function sum_array_fast(arr)
    total = 0
    @inbounds @simd for i in eachindex(arr)
        total += arr[i]
    end
    return total
end

# ⚡ Ускорение: 2-5x!

Совет 6: Избегайте TTFP с PackageCompiler
-----------------------------------------
# Создайте системный образ:
using PackageCompiler
create_sysimage(["YourPackage"], sysimage_path="custom_sys.so")

# Запуск с образом:
julia -J custom_sys.so script.jl

# ⚡ Устраняет задержку первой компиляции!

using JuMP, HiGHS

function test_optimization(n)
    model = Model(HiGHS.Optimizer)
    @variable(model, 0 <= x[1:n] <= 100)
    @constraint(model, sum(x) <= n * 50)
    @objective(model, Max, sum(x))
    optimize!(model)
    return objective_value(model)
end

@benchmark test_optimization(1000)
