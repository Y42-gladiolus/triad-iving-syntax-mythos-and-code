3. ОПТИМИЗАЦИЯ: КАК УСКОРИТЬ V
================================================================================

Совет 1: Используйте -prod для production
-----------------------------------------
# МЕДЛЕННО (debug):
v build app.v

# БЫСТРО (production):
v build -prod app.v

# ⚡ Prod mode ускоряет в 5-10x!

Совет 2: Избегайте лишних аллокаций
-----------------------------------
# МЕДЛЕННО (allocation в loop):
mut arr := []int{}
for i in 0..1000 {
  arr << compute_value()
}

# БЫСТРО (pre-allocate):
mut arr := []int{cap: 1000}
for i in 0..1000 {
  arr << compute_value()
}

# ⚡ Pre-allocation быстрее!

Совет 3: Используйте struct для value types
-------------------------------------------
# МЕДЛЕННО (class-like):
struct Point {
  x int
  y int
}

# БЫСТРО (struct):
struct Point {
  x int
  y int
}

# ⚡ Structs быстрее для value types!

Совет 4: Используйте C interop для performance-critical кода
------------------------------------------------------------
# МЕДЛЕННО (V code):
fn compute() int {
  // Complex computation
}

# БЫСТРО (C code):
#flag -lc
#include <math.h>
unsafe {
  result := C.sqrt(2.0)
}

# ⚡ C code быстрее для critical sections!

Совет 5: Используйте spawn для concurrency
------------------------------------------
# МЕДЛЕННО (последовательно):
for i in 0..10 {
  process(i)
}

# БЫСТРО (параллельно):
for i in 0..10 {
  spawn fn(i int) {
    process(i)
  }(i)
}

# ⚡ Parallel быстрее на multi-core!

9. БЕНЧМАРК: ПРОВЕРЬТЕ САМИ
================================================================================

Создайте benchmark.v:
---------------------
fn sum_enum(n int) i64 {
  mut sum := i64(0)
  for i := 1; i <= n; i++ {
    sum += i
  }
  return sum
}

fn main() {
  start := time.now()
  result := sum_enum(1_000_000_000)
  finish := time.now()
  
  println('Sum: ${result}')
  println('Time: ${finish.unix_millis() - start.unix_millis()} ms')
}

Компиляция и запуск:
--------------------
v build -prod benchmark.v
./benchmark
