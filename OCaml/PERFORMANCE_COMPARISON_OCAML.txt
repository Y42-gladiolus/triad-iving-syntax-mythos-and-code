3. ОПТИМИЗАЦИЯ: КАК УСКОРИТЬ OCAML
================================================================================

Совет 1: Используйте ocamlopt (native) не ocamlc (bytecode)
-----------------------------------------------------------
# МЕДЛЕННО (bytecode):
ocamlc -o program program.ml
# Время выполнения: медленное

# БЫСТРО (native):
ocamlopt -o program program.ml
# Время выполнения: быстрое (5-10x быстрее!)

Совет 2: Tail-Call Optimization
-------------------------------
(* МЕДЛЕННО (не tail-recursive): *)
let rec sum lst =
  match lst with
  | [] -> 0
  | h :: t -> h + sum t  (* Stack растёт! *)

(* БЫСТРО (tail-recursive): *)
let rec sum lst acc =
  match lst with
  | [] -> acc
  | h :: t -> sum t (acc + h)  (* Stack не растёт! *)

(* ⚡ Ускорение: бесконечная рекурсия без переполнения! *)

Совет 3: Избегайте @ для больших списков
----------------------------------------
(* МЕДЛЕННО (O(n²)): *)
let rec concat lists =
  match lists with
  | [] -> []
  | h :: t -> h @ concat t  (* Квадратичное время! *)

(* БЫСТРО (O(n)): *)
let concat lists = List.concat lists

(* ⚡ Ускорение: 10-100x для больших списков! *)

Совет 4: Используйте массивы для случайного доступа
---------------------------------------------------
(* МЕДЛЕННО (O(n) доступ): *)
let lst = [1; 2; 3; 4; 5]
let x = List.nth lst 3  (* O(n) *)

(* БЫСТРО (O(1) доступ): *)
let arr = [|1; 2; 3; 4; 5|]
let x = arr.(3)  (* O(1) *)

(* ⚡ Ускорение: 10-100x для случайного доступа! *)

Совет 5: Включите оптимизации компилятора
-----------------------------------------
# С оптимизациями:
ocamlopt -O2 -o program program.ml

# С профилем:
ocamlopt -p -o program program.ml

# В dune:
(executable
  (name program)
  (flags (:standard -O2)))

(* ⚡ Ускорение: 2-5x с -O2! *)

Совет 6: Избегайте boxed integers для численных вычислений
----------------------------------------------------------
(* МЕДЛЕННО (boxed): *)
let sum lst = List.fold_left (+) 0 lst

(* БЫСТРО (unboxed arrays): *)
let sum_arr arr =
  let total = ref 0 in
  for i = 0 to Array.length arr - 1 do
    total := !total + arr.(i)
  done;
  !total

(* ⚡ Ускорение: 2-5x для интенсивных вычислений! *)

9. БЕНЧМАРК: ПРОВЕРЬТЕ САМИ
================================================================================

Создайте benchmark.ml:
----------------------
let rec fib n =
  if n < 2 then n else fib (n - 1) + fib (n - 2)

let () =
  let start = Sys.time () in
  let result = fib 40 in
  let finish = Sys.time () in
  Printf.printf "Fib(40) = %d\n" result;
  Printf.printf "Time: %.2f seconds\n" (finish -. start)

Компиляция и запуск:
--------------------
ocamlopt -O2 -o benchmark benchmark.ml
./benchmark

Сравнение bytecode vs native:
-----------------------------
ocamlc -o benchmark_byte benchmark.ml
ocamlopt -O2 -o benchmark_native benchmark.ml

time ./benchmark_byte
time ./benchmark_native

(* ⚡ Native будет в 5-10 раз быстрее! *)
