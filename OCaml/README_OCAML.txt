Linux (Ubuntu/Debian):
----------------------
# Шаг 1: Установить opam
sudo apt update
sudo apt install opam

# Шаг 2: Инициализировать opam
opam init --disable-sandboxing  # Если нет docker
# или
opam init  # С sandboxing (рекомендуется)

# Шаг 3: Настроить окружение
eval $(opam env)

# Шаг 4: Установить OCaml
opam switch create 4.14.0  # Или последнюю стабильную
eval $(opam env)

# Проверка
ocaml --version
opam --version

macOS (Homebrew):
-----------------
brew install opam
opam init
eval $(opam env)
opam switch create 4.14.0
eval $(opam env)

----------------------

================================================================================
3. БЫСТРЫЙ СТАРТ
================================================================================

# Запуск REPL (toplevel)
ocaml

# Вы увидите:
# OCaml version 4.14.0
# #

# Первый код:
# let x = 5;;
val x : int = 5

# let square x = x * x;;
val square : int -> int = <fun>

# square 5;;
- : int = 25

# Выход:
# #exit;;
или Ctrl+D

# Запуск скрипта:
ocaml script.ml

# Компиляция в байт-код:
ocamlc -o program program.ml

# Компиляция в нативный код (быстрее!):
ocamlopt -o program program.ml

# С build system (dune - рекомендуется!):
dune init proj my_project
cd my_project
dune build
dune exec ./my_project.exe

================================================================================
4. УПРАВЛЕНИЕ ПАКЕТАМИ (OPAM)
================================================================================

# Поиск пакетов
opam search package-name

# Установка пакета
opam install package-name

# Установка нескольких пакетов
opam install dune merlin ocaml-lsp-server utop

# Просмотр установленных пакетов
opam list

# Обновление пакетов
opam update
opam upgrade

# Удаление пакета
opam remove package-name

# Создание нового проекта
opam init
opam switch create 4.14.0
dune init proj my_project

# Основные команды dune:
dune build          # Собрать проект
dune exec           # Запустить
dune test           # Тесты
dune clean          # Очистить
dune utop           # REPL с проектом

================================================================================
5. ПОДВОДНЫЕ КАМНИ УСТАНОВКИ
================================================================================

⚠️ Проблема 1: opam env не настроен
   Команды ocaml/opam не найдены после установки
   
   Решение: Добавить в ~/.bashrc
   eval $(opam env)

⚠️ Проблема 2: Конфликты версий OCaml
   Разные проекты требуют разные версии
   
   Решение: Использовать opam switch
   opam switch create 4.14.0
   opam switch 4.14.0

⚠️ Проблема 3: Долгая первая компиляция
   Первый opam install может занять 30-60 минут
   
   Решение: Это нормально - компиляция зависимостей
   Используйте opam install -j 4 для параллелизма

⚠️ Проблема 4: utop не запускается
   utop - улучшенный REPL
   
   Решение: 
   opam install utop
   eval $(opam env)

⚠️ Проблема 5: Мерлин не работает в VS Code
   Автодополнение не работает
   
   Решение:
   opam install merlin ocaml-lsp-server
   Перезапустить VS Code

⚠️ Проблема 6: Библиотеки не находятся
   #use "topfind";; не работает
   
   Решение:
   opam install findlib
   #use "topfind";;

8. РЕКОМЕНДУЕМЫЕ ПАКЕТЫ ДЛЯ НАЧАЛА
================================================================================

# Базовый набор (установить сразу):
opam install --batch \
    dune \              # Build system (обязательно!)
    utop \              # Улучшенный REPL
    merlin \            # Автодополнение
    ocaml-lsp-server \  # Language Server
    ocp-indent \        # Форматирование
    ocamlformat \       # Автоформатирование
    
    # Библиотеки
    base \              # Стандартная библиотека (Jane Street)
    core \              # Расширенная стандартная библиотека
    async \             # Асинхронность (Jane Street)
    lwt \               # Альтернативная асинхронность
    yojson \            # JSON парсинг
    ezjsonm \           # Альтернативный JSON
    pprint \            # Красивый вывод
    qcheck \            # Property-based тесты
    odoc \              # Документация
