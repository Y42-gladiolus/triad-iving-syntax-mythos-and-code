3. ОПТИМИЗАЦИЯ: КАК УСКОРИТЬ ODIN
================================================================================

Совет 1: Используйте defer для очистки
--------------------------------------
# МЕДЛЕННО (manual cleanup):
ptr := new(int)
// ...
delete(ptr)

# БЫСТРО (defer):
ptr := new(int)
defer delete(ptr)

# ⚡ Defer гарантирует cleanup!

Совет 2: Используйте stack allocation
-------------------------------------
# МЕДЛЕННО (heap allocation):
ptr := new([100]int)
defer delete(ptr)

# БЫСТРО (stack allocation):
arr: [100]int

# ⚡ Stack быстрее чем heap!

Совет 3: Используйте SIMD для vector operations
-----------------------------------------------
# МЕДЛЕННО (scalar):
for i in 0..<len(data) {
  data[i] = data[i] * 2.0
}

# БЫСТРО (SIMD):
import "core:simd"
for i := 0; i < len(data); i += 4 {
  v := simd.load(&data[i])
  v = v * simd.splat(2.0)
  simd.store(&data[i], v)
}

# ⚡ SIMD в 4-8 раз быстрее!

Совет 4: Используйте SoA для cache-friendly access
--------------------------------------------------
# МЕДЛЕННО (AoS):
Particles :: struct {
  x, y, z: f32,
}
particles: [1000]Particles

# БЫСТРО (SoA):
Particles :: struct {
  x, y, z: [1000]f32,
}

# ⚡ SoA лучше для vectorization!

Совет 5: Используйте multiple return values для error handling
--------------------------------------------------------------
# МЕДЛЕННО (global error):
var last_error: bool
result := divide(a, b)
if last_error { return }

# БЫСТРО (multiple return):
result, ok := divide(a, b)
if !ok { return }

# ⚡ Multiple return values type-safe!

import "core:fmt"

main :: proc() {
  sum: i64 = 0
  for i := 1; i <= 1_000_000_000; i += 1 {
    sum += i
  }
  fmt.println("Sum: ", sum)
}
