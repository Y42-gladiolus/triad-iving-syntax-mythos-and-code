3. ОПТИМИЗАЦИЯ: КАК УСКОРИТЬ HARE
================================================================================

Совет 1: Используйте defer для очистки
--------------------------------------
# МЕДЛЕННО (manual cleanup):
let f = posix::open("file.txt", posix::O_RDONLY);
// ...
posix::close(f);

# БЫСТРО (defer):
let f = posix::open("file.txt", posix::O_RDONLY);
defer posix::close(f);

# ⚡ Defer гарантирует cleanup!

Совет 2: Используйте stack allocation
-------------------------------------
# МЕДЛЕННО (heap allocation):
let ptr = posix::malloc(100);
defer posix::free(ptr);

# БЫСТРО (stack allocation):
let arr: [100]u8 = [0, ...];

# ⚡ Stack быстрее чем heap!

Совет 3: Используйте slices вместо pointers
-------------------------------------------
# МЕДЛЕННО (pointer + length):
let ptr: *u8 = data;
let len: usize = 100;

# БЫСТРО (slice):
let slice: []u8 = data[..100];

# ⚡ Slices safer и проще!

Совет 4: Используйте match вместо if-else
-----------------------------------------
# МЕДЛЕННО (nested if):
if (x == 1) {
  // ...
} else if (x == 2) {
  // ...
} else {
  // ...
};

# БЫСТРО (match):
match (x) {
case 1 =>
  // ...
case 2 =>
  // ...
case =>
  // ...
};

# ⚡ Match clearer и быстрее!

Совет 5: Используйте POSIX APIs напрямую
----------------------------------------
# МЕДЛЕННО (wrapper libraries):
// Extra abstraction layer

# БЫСТРО (direct POSIX):
posix::open(...);
posix::read(...);
posix::write(...);

# ⚡ Direct APIs быстрее!
