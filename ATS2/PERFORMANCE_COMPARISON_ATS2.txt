┌──────────────────────────────────────────────────────────────────────────────┐
│                    РЕАЛЬНАЯ ПРОИЗВОДИТЕЛЬНОСТЬ ATS2                          │
├──────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│  МИФ: "Безопасность памяти требует накладных расходов"                       │
│  РЕАЛЬНОСТЬ: ATS2 доказывает безопасность на compile-time, не runtime!      │
│                                                                              │
│  ⚡ Где ATS2 БЫСТРЫЙ:                                                        │
│  • Compile-time проверки (не runtime оверхед)                               │
│  • Компиляция в чистый C (оптимизации GCC/Clang)                            │
│  • Нет GC пауз (ручное управление памятью)                                  │
│  • Linear types = нулевой оверхед безопасности                              │
│  • Встраивание в C код (без runtime)                                        │
│                                                                              │
│  🐌 Где ATS2 МЕДЛЕННЫЙ:                                                      │
│  • Время компиляции (проверка типов сложная)                                │
│  • Разработка медленнее (нужно писать доказательства)                       │
│  • Меньше библиотек чем C/C++                                               │
│  • Кривая обучения очень крутая                                             │
│                                                                              │
│  ⚡ Erasure: доказательства удаляются при компиляции!                       │
│                                                                              │
└──────────────────────────────────────────────────────────────────────────────┘

3. ОПТИМИЗАЦИЯ: КАК УСКОРИТЬ ATS2
================================================================================

Совет 1: Используйте --optimize для компиляции
----------------------------------------------
# МЕДЛЕННО (без оптимизаций):
patscc -o program program.dats

# БЫСТРО (с оптимизациями):
patscc -O2 -o program program.dats

# ⚡ Ускорение: 5-10x!

Совет 2: Избегайте доказательств в runtime коде
-----------------------------------------------
(* МЕДЛЕННО (доказательство в runtime): *)
fun process (x: int) (pf: proof x > 0): int = ...

(* БЫСТРО (доказательство в типе): *)
fun process {n:int | n > 0} (x: int n): int = ...

(* ⚡ Доказательство проверяется на compile-time, не runtime! *)

Совет 3: Используйте inline для маленьких функций
-------------------------------------------------
(* Без inline: *)
fun add (x: int, y: int): int = x + y

(* С inline: *)
fun add (x: int, y: int): int = x + y
implement add (x, y) = x + y

(* ⚡ Ускорение: 2-3x для маленьких функций! *)

Совет 4: Избегайте box для маленьких структур
---------------------------------------------
(* МЕДЛЕННО (heap allocation): *)
val x: int = $box (5)

(* БЫСТРО (stack allocation): *)
val x: int = 5

(* ⚡ Ускорение: 10-100x для частых аллокаций! *)

Совет 5: Используйте C функции для критичного кода
--------------------------------------------------
(* ATS2 код: *)
fun sum (n: int): int = ...

(* C код (быстрее для численных вычислений): *)
%{
  int sum_c(int n) {
    return n * (n + 1) / 2;
  }
%}

(* ⚡ Ускорение: 2-5x для численных вычислений! *)

9. БЕНЧМАРК: ПРОВЕРЬТЕ САМИ
================================================================================

Создайте benchmark.dats:
------------------------
#include "share/atspre_staload.hats"

fun sumTo (n: int): int =
  if n > 0 then n + sumTo (n - 1) else 0

implement main0 () =
  val start = clock_now ()
  val result = sumTo (1000000)
  val finish = clock_now ()
  val () = print ("Sum: ")
  val () = print_int (result)
  val () = print_newline ()
  val () = print ("Time: ")
  val () = print_int (gclock_diff (finish, start))
  val () = print (" ms\n")

Компиляция и запуск:
--------------------
patscc -O2 -o benchmark benchmark.dats
./benchmark
