// Базовая функция
int add(int a, int b) {
    return a + b;
}

// Void функция
void print_hello() {
    printf("Hello\n");
}

// Pointer параметры
void increment(int* x) {
    (*x)++;
}

// Array параметры
void print_array(int arr[], int size) {
    for (int i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }
}

// Function pointers
int add(int a, int b) { return a + b; }
int (*func_ptr)(int, int) = &add;
int result = func_ptr(5, 3);  // 8

// Inline функция (C99)
inline int square(int x) {
    return x * x;
}

// Variadic функции
#include <stdarg.h>
int sum(int count, ...) {
    va_list args;
    va_start(args, count);
    int total = 0;
    for (int i = 0; i < count; i++) {
        total += va_arg(args, int);
    }
    va_end(args);
    return total;
}

// Recursive функции
int factorial(int n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

    // Handle error

    int* ptr = malloc(sizeof(int));

10. ТИПИЧНЫЕ ОШИБКИ (РАСШИРЕНО!)
================================================================================

❌ Ошибка 1: Memory leak
int* ptr = malloc(sizeof(int) * 100);
// Забыли free!

✓ Решение:
int* ptr = malloc(sizeof(int) * 100);
// ...
free(ptr);
ptr = NULL;

❌ Ошибка 2: Buffer overflow
char buffer[10];
strcpy(buffer, "This is too long!");  // ОШИБКА!

✓ Решение:
char buffer[10];
strncpy(buffer, "This is too long!", sizeof(buffer) - 1);
buffer[sizeof(buffer) - 1] = '\0';

❌ Ошибка 3: Dangling pointer
int* ptr = malloc(sizeof(int));
free(ptr);
*ptr = 5;  // ОШИБКА! Use after free

✓ Решение:
int* ptr = malloc(sizeof(int));
// ...
free(ptr);
ptr = NULL;  // Prevent use after free

❌ Ошибка 4: Null pointer dereference
int* ptr = NULL;
*ptr = 5;  // ОШИБКА! Segmentation fault

✓ Решение:
int* ptr = NULL;
if (ptr != NULL) {
    *ptr = 5;
}

❌ Ошибка 5: Array out of bounds
int arr[5];
arr[10] = 5;  // ОШИБКА! Out of bounds

✓ Решение:
int arr[5];
for (int i = 0; i < 5; i++) {
    arr[i] = i;
}

❌ Ошибка 6: Uninitialized variable
int x;
printf("%d\n", x);  // ОШИБКА! Undefined behavior

✓ Решение:
int x = 0;
printf("%d\n", x);

❌ Ошибка 7: Wrong format specifier
int x = 5;
printf("%f\n", x);  // ОШИБКА! %f для float

✓ Решение:
int x = 5;
printf("%d\n", x);

❌ Ошибка 8: Pointer arithmetic error
int arr[5] = {1, 2, 3, 4, 5};
int* ptr = arr;
printf("%d\n", ptr[5]);  // ОШИБКА! Out of bounds

✓ Решение:
int arr[5] = {1, 2, 3, 4, 5};
int* ptr = arr;
for (int i = 0; i < 5; i++) {
    printf("%d\n", ptr[i]);
}

❌ Ошибка 9: Const correctness
const int x = 5;
x = 10;  // ОШИБКА! Cannot modify const

✓ Решение:
int x = 5;
x = 10;  // OK

❌ Ошибка 10: Return local pointer
int* get_array() {
    int arr[10];
    return arr;  // ОШИБКА! arr на стеке
}

✓ Решение:
int* get_array() {
    int* arr = malloc(10 * sizeof(int));
    return arr;  // OK (caller must free)
}

Пример 1: Linked list
---------------------
typedef struct Node {
    int data;
    struct Node* next;
} Node;

Node* create_node(int data) {
    Node* node = malloc(sizeof(Node));
    node->data = data;
    node->next = NULL;
    return node;
}

void free_list(Node* head) {
    while (head != NULL) {
        Node* temp = head;
        head = head->next;
        free(temp);
    }
}

Пример 2: Dynamic array
-----------------------
typedef struct {
    int* data;
    size_t size;
    size_t capacity;
} DynamicArray;

DynamicArray* create_array() {
    DynamicArray* arr = malloc(sizeof(DynamicArray));
    arr->capacity = 10;
    arr->size = 0;
    arr->data = malloc(arr->capacity * sizeof(int));
    return arr;
}

void push(DynamicArray* arr, int value) {
    if (arr->size >= arr->capacity) {
        arr->capacity *= 2;
        arr->data = realloc(arr->data, arr->capacity * sizeof(int));
    }
    arr->data[arr->size++] = value;
}

Пример 3: File copy
-------------------
void copy_file(const char* src, const char* dest) {
    FILE* src_file = fopen(src, "rb");
    FILE* dest_file = fopen(dest, "wb");
    
    char buffer[1024];
    size_t bytes;
    
    while ((bytes = fread(buffer, 1, sizeof(buffer), src_file)) > 0) {
        fwrite(buffer, 1, bytes, dest_file);
    }
    
    fclose(src_file);
    fclose(dest_file);
}

Пример 4: String tokenizer
--------------------------
#include <string.h>

char* tokenize(char* str, const char* delim) {
    static char* static_str = NULL;
    
    if (str != NULL) {
        static_str = str;
    }
    
    if (static_str == NULL) {
        return NULL;
    }
    
    char* token = strtok(static_str, delim);
    static_str = NULL;
    
    return token;
}
