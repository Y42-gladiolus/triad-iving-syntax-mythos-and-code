3. ОПТИМИЗАЦИЯ: КАК УСКОРИТЬ RE SCRIPT
================================================================================

Совет 1: Используйте watch режим для разработки
-----------------------------------------------
# МЕДЛЕННО (полная компиляция каждый раз):
rescript build

# БЫСТРО (incremental compilation):
rescript build -w

# ⚡ Ускорение: 5-10x для разработки!

Совет 2: Избегайте JS интероперабельности в hot path
----------------------------------------------------
# МЕДЛЕННО (JS вызовы):
@bs.val
external jsFunction: int => int = "jsFunction"

# БЫСТРО (native ReScript):
let nativeFunction = x => x * 2

# ⚡ Native ReScript быстрее JS вызовов!

Совет 3: Используйте immutable структуры
----------------------------------------
# МЕДЛЕННО (mutable в циклах):
let mutable sum = 0
for (i in 1 to 1000) {
  sum = sum + i
}

# БЫСТРО (functional):
let sum = List.foldLeft((acc, x) => acc + x, 0, numbers)

# ⚡ ReScript оптимизирует immutable код!

Совет 4: Используйте pattern matching вместо if/else
----------------------------------------------------
# МЕДЛЕННО (вложенные if):
if (condition1) {
  if (condition2) {
    result1
  } else {
    result2
  }
}

# БЫСТРО (pattern matching):
switch (condition1, condition2) {
| (true, true) => result1
| (true, false) => result2
| _ => result3
}

# ⚡ Pattern matching оптимизирован компилятором!

Совет 5: Используйте Belt утилиты
---------------------------------
# МЕДЛЕННО (ручная обработка Option):
switch user {
| Some(u) => u.name
| None => "Unknown"
}

# БЫСТРО (Belt утилиты):
Option.getWithDefault(user, {name: "Unknown"})

# ⚡ Belt оптимизирован для ReScript!

type user = {name: string, age: int}

let createUser = (name, age) => {name, age}

9. БЕНЧМАРК: ПРОВЕРЬТЕ САМИ
================================================================================

Создайте Benchmark.res:
-----------------------
let rec sumEnum = n => {
  if (n <= 0) {
    0
  } else {
    n + sumEnum(n - 1)
  }
}

let start = Js.Date.now()
let result = sumEnum(10000000)
let finish = Js.Date.now()

Js.log("Sum:", result)
Js.log("Time:", finish - start, "ms")

Компиляция и запуск:
--------------------
rescript build
node lib/Benchmark.js

Сравнение с TypeScript:
-----------------------
# ReScript будет быстрее на компиляции
# Runtime одинаковый (оба JS)
