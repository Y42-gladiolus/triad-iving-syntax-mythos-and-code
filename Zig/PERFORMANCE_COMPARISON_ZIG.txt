3. ОПТИМИЗАЦИЯ: КАК УСКОРИТЬ ZIG
================================================================================

Совет 1: Используйте ReleaseFast для production
-----------------------------------------------
// МЕДЛЕННО (Debug):
zig build

// БЫСТРО (ReleaseFast):
zig build -Doptimize=ReleaseFast

// ⚡ Ускорение: 5-10x для runtime!

Совет 2: Используйте comptime для вычислений
--------------------------------------------
// МЕДЛЕННО (runtime):
fn square(x: i32) i32 {
    return x * x;
}

// БЫСТРО (comptime):
fn square(comptime x: comptime_int) comptime_int {
    return x * x;
}

// ⚡ Вычисляется при компиляции!

Совет 3: Избегайте аллокаций в циклах
-------------------------------------
// МЕДЛЕННО (аллокация в цикле):
for (items) |item| {
    const ptr = allocator.alloc(u8, 10);
    defer allocator.free(ptr);
}

// БЫСТРО (одна аллокация):
const ptr = allocator.alloc(u8, 10 * items.len);
defer allocator.free(ptr);

// ⚡ Меньше аллокаций = быстрее!

Совет 4: Используйте slice вместо pointer
-----------------------------------------
// МЕДЛЕННО (pointer + длина отдельно):
fn process(ptr: [*]u8, len: usize) void {
    // ...
}

// БЫСТРО (slice):
fn process(slice: []u8) void {
    // ...
}

// ⚡ Безопаснее и удобнее!

Совет 5: Inline функции для hot path
------------------------------------
// Обычная функция:
fn add(a: i32, b: i32) i32 {
    return a + b;
}

// Inline функция:
inline fn add(a: i32, b: i32) i32 {
    return a + b;
}

// ⚡ Встраивается в код (нет вызова функции)!

9. БЕНЧМАРК: ПРОВЕРЬТЕ САМИ
================================================================================

Создайте benchmark.zig:
-----------------------
const std = @import("std");

pub fn main() !void {
    var sum: i64 = 0;
    const start = try std.time.Instant.now();
    
    var i: i64 = 0;
    while (i < 1_000_000_000) : (i += 1) {
        sum += i;
    }
    
    const end = try std.time.Instant.now();
    const ms = end.since(start) / std.time.ns_per_ms;
    
    std.debug.print("Sum: {}\n", .{sum});
    std.debug.print("Time: {} ms\n", .{ms});
}

Компиляция и запуск:
--------------------
zig build-exe benchmark.zig -O ReleaseFast
./benchmark
