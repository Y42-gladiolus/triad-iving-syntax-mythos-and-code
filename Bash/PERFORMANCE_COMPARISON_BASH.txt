3. ОПТИМИЗАЦИЯ: КАК УСКОРИТЬ BASH
================================================================================

Совет 1: Используйте встроенные команды
---------------------------------------
# МЕДЛЕННО (внешние команды):
cat file.txt | grep pattern | wc -l

# БЫСТРО (встроенные):
grep -c pattern file.txt

# ⚡ Меньше процессов = быстрее!

Совет 2: Избегайте subshells
----------------------------
# МЕДЛЕННО (subshell):
COUNT=$(cat file.txt | wc -l)

# БЫСТРО (встроенные):
COUNT=$(wc -l < file.txt)

# ⚡ Subshells имеют оверхед!

Совет 3: Используйте case вместо if
-----------------------------------
# МЕДЛЕННО (multiple if):
if [ "$VAR" = "start" ]; then
    echo "start"
elif [ "$VAR" = "stop" ]; then
    echo "stop"
elif [ "$VAR" = "restart" ]; then
    echo "restart"
fi

# БЫСТРО (case):
case "$VAR" in
    start|run) echo "start" ;;
    stop|quit) echo "stop" ;;
    restart) echo "restart" ;;
    *) echo "Unknown" ;;
esac

# ⚡ case эффективнее для множественных проверок!

Совет 4: Минимизируйте external commands
----------------------------------------
# МЕДЛЕННО (external):
DIRNAME=$(dirname "$FILE")
BASENAME=$(basename "$FILE")

# БЫСТРО (parameter expansion):
DIRNAME="${FILE%/*}"
BASENAME="${FILE##*/}"

# ⚡ Parameter expansion быстрее!

Совет 5: Используйте set -euo pipefail
--------------------------------------
# МЕДЛЕННО (проверка каждой команды):
command1
if [ $? -ne 0 ]; then
    exit 1
fi
command2

# БЫСТРО (set -euo pipefail):
set -euo pipefail
command1
command2

# ⚡ Автоматическая обработка ошибок!

Совет 6: Используйте arrays вместо строк
----------------------------------------
# МЕДЛЕННО (string parsing):
LIST="item1 item2 item3"
for item in $LIST; do
    echo $item
done

# БЫСТРО (arrays):
ARRAY=(item1 item2 item3)
for item in "${ARRAY[@]}"; do
    echo $item
done

# ⚡ Arrays надёжнее!
