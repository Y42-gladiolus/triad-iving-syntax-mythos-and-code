3. ОПТИМИЗАЦИЯ: КАК УСКОРИТЬ RUST
================================================================================

Совет 1: Используйте release профиль
------------------------------------
# МЕДЛЕННО (debug):
cargo build
# Время компиляции: быстрое
# Время выполнения: медленное

# БЫСТРО (release):
cargo build --release
# Время компиляции: медленное
# Время выполнения: быстрое (10-50x!)

Совет 2: Включите LTO (Link Time Optimization)
----------------------------------------------
# Cargo.toml:
[profile.release]
lto = true
codegen-units = 1

# ⚡ Ускорение: 10-30% для CPU-bound задач

Совет 3: Используйте rayon для параллелизма
-------------------------------------------
// МЕДЛЕННО (последовательно):
let sum: i32 = (0..1000000)
    .map(|x| x * x)
    .sum();

// БЫСТРО (параллельно):
use rayon::prelude::*;
let sum: i32 = (0..1000000)
    .into_par_iter()
    .map(|x| x * x)
    .sum();

# ⚡ Ускорение: 4-8x на 8 ядрах!

Совет 4: Избегайте лишних аллокаций
-----------------------------------
// МЕДЛЕННО (аллокация в цикле):
let mut result = Vec::new();
for i in 0..1000 {
    result.push(i * 2);  // Может realloc!
}

// БЫСТРО (предварительное выделение):
let mut result = Vec::with_capacity(1000);
for i in 0..1000 {
    result.push(i * 2);
}

// ⚡ Ускорение: 2-5x!

Совет 5: Используйте &str вместо String для параметров
------------------------------------------------------
// МЕДЛЕННО (клонирование):
fn process(s: String) {
    // ...
}
process(my_string.clone());

// БЫСТРО (заимствование):
fn process(s: &str) {
    // ...
}
process(&my_string);

# ⚡ Ускорение: избегает клонирования!

Совет 6: Профилируйте с flamegraph
----------------------------------
cargo install flamegraph
cargo flamegraph --bin my_program

# Покажет горячие точки для оптимизации

Rust + Python (PyO3):
---------------------
// Cargo.toml: [lib] crate-type = ["cdylib"]
use pyo3::prelude::*;

#[pyfunction]
fn fibonacci(n: u32) -> u64 {
    // Быстрая реализация на Rust
}

#[pymodule]
fn my_module(_py: Python, m: &PyModule) -> PyResult<()> {
    m.add_function(wrap_pyfunction!(fibonacci, m)?)?;
    Ok(())
}

# ⚡ Вызов из Python как обычный модуль!

Rust + R (extendr):
-------------------
// Cargo.toml: [lib] crate-type = ["staticlib"]
use extendr_api::prelude::*;

#[extendr]
fn fast_sum(x: &[f64]) -> f64 {
    x.iter().sum()
}

# ⚡ Вызов из R как обычная функция!

Rust + MiniZinc (через CLI):
----------------------------
use std::process::Command;

fn run_minizinc(model: &str, data: &str) -> String {
    let output = Command::new("minizinc")
        .args(["--solver", "CBC", model, data])
        .output()
        .expect("Failed to run minizinc");
    
    String::from_utf8_lossy(&output.stdout).to_string()
}

# ⚡ Rust как обёртка для MiniZinc в production!

8. БЕНЧМАРК: ПРОВЕРЬТЕ САМИ
================================================================================

Создайте Cargo.toml:
--------------------
[package]
name = "benchmark"
version = "0.1.0"
edition = "2021"

[dependencies]
rayon = "1.8"

[dev-dependencies]
criterion = "0.5"

[[bench]]
name = "my_benchmark"
harness = false

Создайте benches/my_benchmark.rs:
---------------------------------
use criterion::{black_box, criterion_group, criterion_main, Criterion};
use rayon::prelude::*;

fn sequential_sum(n: u64) -> u64 {
    (0..n).sum()
}

fn parallel_sum(n: u64) -> u64 {
    (0..n).into_par_iter().sum()
}

fn benchmark_sums(c: &mut Criterion) {
    c.bench_function("sequential 1e9", |b| {
        b.iter(|| sequential_sum(black_box(1_000_000_000)))
    });
    c.bench_function("parallel 1e9", |b| {
        b.iter(|| parallel_sum(black_box(1_000_000_000)))
    });
}

criterion_group!(benches, benchmark_sums);
criterion_main!(benches);

Запуск:
-------
cargo bench
